الاستخدام

USAGE · TOKENS IN CODE

جميع التوكنز متاحة عبر window.MESA في كل ملفات jsx.

// tokens.jsx — central design tokens
const MESA = {
  // colors
  burgundy:     '#8B1A1A',
  gold:         '#C9A961',
  cream:        '#FAF7F2',
  ink:          '#1A1A1A',
  muted:        '#6B5E52',
  border:       '#E8DFD1',
  olive:        '#6B7F3F',
  amber:        '#D97706',
  danger:       '#991B1B',

  // type stacks
  arabic: '"Tajawal", system-ui',
  sans:   '"Inter", -apple-system',
  mono:   '"IBM Plex Mono"',
};
// example usage in a screen
function DishCard({ dish }) {
  return (
    <div style={{
      background: MESA.surface,
      border: `1px solid ${MESA.borderLite}`,
      borderRadius: 14,
      padding: 14,
    }}>
      <DishGlyph dish={dish} size={56} />
      <div style={{
        fontFamily: MESA.arabic,
        fontSize: 15,
        fontWeight: 600,
        color: MESA.ink,
      }}>{dish.ar}</div>
      <div style={{
        color: MESA.burgundy,
        fontVariantNumeric: 'tabular-nums',
      }}>{iqd(dish.price)}</div>
    </div>
  );
}
